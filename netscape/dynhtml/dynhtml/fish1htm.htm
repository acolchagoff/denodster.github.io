	<SCRIPT>

	<!-- Simple move function -->
	function movefish() {
		var fish = document.layers["fish"];
		if (fish.left < 400) {
			fish.offset(5, 0);}
		else {fish.left = 10;}
		setTimeout("movefish()", 10);
   		return;
	}

	</SCRIPT>

	<H1>Fish Example 1</H1>
	<FORM>
	<INPUT type=button value="Move the fish" OnClick="movefish(); return false;">
